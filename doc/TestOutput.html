def limit_to_rows(number_of_rows)
  output = TestOutput.new("")
  raws_count = 1;
  self.lines.each do |current_line|
    output += current_line
    break if(raws_count == number_of_rows)
    raws_count+=1
  end
  output=output+"..." if ( number_of_rows < self.lines.count )
  output
end
